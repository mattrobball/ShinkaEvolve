R = QQ[x,y,z]

-- Define the ideal of a point configuration
-- Using the singular locus of the B3 reflection arrangement (9 lines)
-- Lines: x, y, z, x+/-y, x+/-z, y+/-z
-- This arrangement creates points with high multiplicity (up to 4 lines intersecting)
-- Such configurations (rational points of high multiplicity arrangements) often have high resurgence
F = x*y*z*(x^2-y^2)*(x^2-z^2)*(y^2-z^2)
J = ideal jacobian ideal F
pts = minimalPrimes J

-- The ideal I is the intersection of the point ideals
I = intersect pts

R = QQ[x,y,z]

-- Define the ideal of a point configuration
-- Using a Star Configuration of 5 generic lines in P^2.
-- The points are the pairwise intersections of these lines (10 points).
-- Such configurations have known high resurgence (rho = 1.6).
-- This should satisfy I^(3) not contained in I^2.

-- 5 generic linear forms
L = {
    x,
    y,
    z,
    x + y + z,
    x + 2*y + 4*z
}

-- The ideal of a star configuration of s lines is generated by all products of s-1 forms.
-- Here s=5, so we take products of 4 forms.
s = 5
gensI = apply(subsets(L, s-1), S -> product S)
I = ideal gensI
